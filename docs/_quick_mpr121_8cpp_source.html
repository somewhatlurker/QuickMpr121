31 // Reads bytes from consecutive MPR121 registers, starting at addr.
32 // Max count is equal to the MPR121_I2C_BUFLEN define.
33 byte* mpr121::readRegister(mpr121Register addr, byte count) {
34  if (count > MPR121_I2C_BUFLEN)
35  count = MPR121_I2C_BUFLEN;
36 
37  // write the address to read from
38  i2cWire->beginTransmission(i2cAddr);
39  i2cWire->write(addr);
40  i2cWire->endTransmission(false); // use false to restart instead of stopping
41 
42  i2cWire->requestFrom(i2cAddr, count, (byte)true); // sendStop is true by defaualt where supported, but setting it guarantees support
43 
44  byte readnum = 0;
45  while (i2cWire->available() && readnum < count)
46  {
47  i2cReadBuf[readnum] = i2cWire->read();
48  readnum++;
49  }
50 
51  // if not fully read, clear additional bytes to avoid returning old data
52  for ( ; readnum < count; readnum++)
53  {
54  i2cReadBuf[readnum] = 0;
55  }
56 
57  return i2cReadBuf;
58 }

200 // Sets "Filter Configuration".
201 // (AN3890)
202 //
203 // FFI: "First Filter Iterations"
204 // CDC: Global "Charge Discharge Current" (μA), not used if autoconfig is enabled (Max 63, Default 16)
205 // CDT: Global "Charge Discharge Time" (μs), not used if autoconfig is enabled
206 // SFI: "Second Filter Iterations"
207 // ESI: "Electrode Sample Interval" (ms)
208 //
209 // Response time is equal to SFI * ESI in ms.
210 void mpr121::setFilterConfig(mpr121FilterFFI FFI, byte CDC, mpr121FilterCDT CDT, mpr121FilterSFI SFI, mpr121FilterESI ESI) {
211  byte FFI_2 = (byte)FFI & 0b00000011;
212  byte CDC_6 = CDC & 0b00111111;
213  byte CDT_3 = (byte)CDT & 0b00000111;
214  byte SFI_2 = (byte)SFI & 0b00000011;
215  byte ESI_3 = (byte)ESI & 0b00000111;
216 
217  writeRegister(MPRREG_FILTER_GLOBAL_CDC_CONFIG, (FFI_2 << 6) | CDC_6);
218  writeRegister(MPRREG_FILTER_GLOBAL_CDT_CONFIG, (CDT_3 << 5) | (SFI_2 << 3) | ESI_3);
219 
220  // update FFI in autoconf
221  byte autoConf = readRegister(MPRREG_AUTOCONFIG_CONTROL_0) & 0b00111111;
222  autoConf |= (FFI_2 << 6);
223  writeRegister(MPRREG_AUTOCONFIG_CONTROL_0, autoConf);
224 }

242 // Sets "Auto-Configure" settings.
243 // (AN3889/datasheet)
244 //
245 // USL: "Up-Side Limit" -- Calculate this as `256L * (supplyMillivolts - 700) / supplyMillivolts`. If unsure, use the value for 1.8V supply (156).
246 // LSL: "Low-Side Limit" -- Calculate this as `USL * 0.65`. If unsure, use the value for 1.8V supply (101).
247 // TL: "Target Level" -- Calculate this as `USL * 0.9`. If unsure, use the value for 1.8V supply (140).
248 // RETRY: Number of retries for failed config before setting the out of range flag.
249 // BVA: "Baseline Value Adjust" -- Changes how the baseline registers will be set after auto-configuration completes
250 // ARE: "Automatic Reconfiguration Enable" -- Out of range (failed) channels will be reconfigured every sampling interval
251 // ACE: "Automatic Configuration Enable" -- Perform auto-configuration when entering run mode
252 // SCTS: "Skip Charge Time Search" -- Skip CDT search and use the already-set CDTx or global CDT. This results in a shorter time to configure, but the designer must supply appropriate values.
253 // OORIE: "Out-of-range interrupt enable" -- Trigger an interrupt when a channel is determined to be out of range
254 // ARFIE: "Auto-reconfiguration fail interrupt enable" -- Trigger an interrupt when auto-reconfiguration fails
255 // ACFIE: "Auto-configuration fail interrupt enable" -- Trigger an interrupt when auto-configuration fails
256 void mpr121::setAutoConfig(byte USL, byte LSL, byte TL, mpr121AutoConfigRetry RETRY, mpr121AutoConfigBVA BVA, bool ARE, bool ACE, bool SCTS, bool OORIE, bool ARFIE, bool ACFIE) {
257  byte FFI = (readRegister(MPRREG_AFE_CONFIG) >> 6) & 0b00000011;
258  byte RETRY_2 = RETRY & 0b00000011;
259  byte BVA_2 = BVA & 0b00000011;
260 
261  writeRegister(MPRREG_AUTOCONFIG_USL, USL);
262  writeRegister(MPRREG_AUTOCONFIG_LSL, LSL);
263  writeRegister(MPRREG_AUTOCONFIG_TL, TL);
264  writeRegister(MPRREG_AUTOCONFIG_CONTROL_0, (FFI << 6) | (RETRY_2 << 4) | (BVA_2 << 2) | ((ARE ? 1 : 0) << 1) | (ACE ? 1 : 0));
265 
266  writeRegister(MPRREG_AUTOCONFIG_CONTROL_1, ((SCTS ? 1 : 0) << 7) | ((OORIE ? 1 : 0) << 2) | ((ARFIE ? 1 : 0) << 1) | (ACFIE ? 1 : 0));
267 }

270 // Sets the GPIO PWM value for consecutive pins.
271 // (AN3894)
272 //
273 // Max value is 15
274 // Pin 9 apparently has a logic bug and pin 10 must also have its data set high for it to work.
275 // (https://community.nxp.com/thread/305474)
276 void mpr121::setPWM(byte pin, byte count, byte value) {
277  if (!checkGPIOPinNum(pin, count))
278  return;
279 
280  pin -= 4; // easier to make it 0-indexed now
281 
282  byte value_4 = value & 0b1111;
283 
284  mpr121Register reg = MPRREG_PWM_DUTY_0;
285  byte regVal = 0;
286 
287  for (byte i = 0; i < count; i++) {
288  if (i == 0 || (pin + i) % 2 == 0) { // if just starting or moving to a new register's start
289  reg = (mpr121Register)(MPRREG_PWM_DUTY_0 + (pin + i)/2);
290  regVal = readRegister(reg);
291  }
292 
293  if ((pin + i) % 2 == 0)
294  regVal = (regVal & 0b11110000) | value_4;
295  else
296  regVal = (regVal & 0b00001111) | (value_4 << 4);
297 
298  if ((pin + i) % 2 == 1 || i == count - 1) // if the last of a register or the last iteration
299  writeRegister(reg, regVal);
300  }
301 }

304 // Creates an MPR121 device with sane default settings.
305 // addr: The I2C address to use. If not specified (or ==0), the next valid address will be chosen automatically.
306 //wire: You can pass in an alternative TwoWire instance.
307 mpr121::mpr121(byte addr, TwoWire *wire)
308 {
309  static byte usedAddresses = 0;
310 
311  if (addr == 0) {
312  if (!bitRead(usedAddresses, 0)) {
313  addr = 0x5a;
314  }
315  else if (!bitRead(usedAddresses, 1)) {
316  addr = 0x5b;
317  }
318  else if (!bitRead(usedAddresses, 2)) {
319  addr = 0x5c;
320  }
321  else if (!bitRead(usedAddresses, 3)) {
322  addr = 0x5d;
323  }
324  else {
325  addr = 0x5a; // at least fall back to *an* mpr121 if all addresses are taken
326  }
327  }
328 
329  if (addr >= 0x5a && addr <= 0x5d)
330  bitSet(usedAddresses, 0x5a - addr);
331 
332  i2cAddr = addr;
333  i2cWire = wire;

346  // FDLtouched = 0x00; // ?

426  if (!checkElectrodeNum(electrode))
427  return false;

431  electrodeTouchCacheMicros = microsNow;
432  #if MPR121_USE_BITFIELDS
433  electrodeTouchCache = readTouchState();
434  #else
435  // this line actually isn't needed, because the data would already be in electrodeTouchBuf after calling readTouchState()

437  readTouchState();
438  #endif
439  }
440 
441  #if MPR121_USE_BITFIELDS
442  return bitRead(electrodeTouchCache, electrode);

585 // Sets pin mode for consecutive GPIO pins.
586 // GPIO can be used on pins 4-11 when they aren't used for sensing.
587 // Use mode MPR_GPIO_MODE_OUTPUT_OPENDRAIN_HIGH for direct LED driving -- it can source up to 12mA.
588 void mpr121::setGPIOMode(byte pin, byte count, mpr121GPIOMode mode) {
589  if (!checkGPIOPinNum(pin, count))
590  return;
591 
592  pin -= 4; // easier to make it 0-indexed now
593 
594 
595  byte enableByte = readRegister(MPRREG_GPIO_ENABLE);
596 
597  // disable the modified outputs while changing stuff around
598  for (byte i = 0; i < count; i++) {
599  bitClear(enableByte, pin + i);
600  }
601  writeRegister(MPRREG_GPIO_ENABLE, enableByte);
602 
603  if (mode == MPR_GPIO_MODE_DISABLED)
604  return; // all done, no need to worry about other values
605 
606 
607  byte tempByte;
608  bool tempVal;
609 
610  // set direction
611  tempByte = readRegister(MPRREG_GPIO_DIRECTION);
612  tempVal = bitRead(mode, 2); // settings are bit-packed into the enum to save doing lookups
613  for (byte i = 0; i < count; i++) {
614  bitWrite(tempByte, pin + i, tempVal);
615  }
616  writeRegister(MPRREG_GPIO_DIRECTION, tempByte);
617 
618  // set control 0 (enable internal resistors or open drain mode)
619  tempByte = readRegister(MPRREG_GPIO_CONTROL_0);
620  tempVal = bitRead(mode, 1);
621  for (byte i = 0; i < count; i++) {
622  bitWrite(tempByte, pin + i, tempVal);
623  }
624  writeRegister(MPRREG_GPIO_CONTROL_0, tempByte);
625 
626  // set control 1 (internal resistor or open drain level)
627  tempByte = readRegister(MPRREG_GPIO_CONTROL_1);
628  tempVal = bitRead(mode, 0);
629  for (byte i = 0; i < count; i++) {
630  bitWrite(tempByte, pin + i, tempVal);
631  }
632  writeRegister(MPRREG_GPIO_CONTROL_1, tempByte);
633 
634 
635  // set enable to final value
636  tempVal = bitRead(mode, 3);
637  for (byte i = 0; i < count; i++) {
638  bitWrite(enableByte, pin + i, tempVal);
639  }
640  writeRegister(MPRREG_GPIO_ENABLE, enableByte);
641 }

743  mpr121::setElectrodeThresholds(i, touchThresholds[i], releaseThresholds[i]);
744  }
